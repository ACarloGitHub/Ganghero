import subprocess
import tempfile

def run(patch: str):
    with tempfile.NamedTemporaryFile(delete=False) as f:
        f.write(patch.encode())
        name = f.name

    r = subprocess.run(
        ["git", "apply", name],
        capture_output=True,
        text=True
    )

    return r.stdout + r.stderr

10. Loop dell’agente
agent/agent_loop.py

from planner import plan
from memory import load_memory, save_memory

from tools.read_file import run as read_file
from tools.write_file import run as write_file
from tools.search_code import run as search_code
from tools.run_terminal import run as run_terminal
from tools.apply_patch import run as apply_patch

TOOLS = {
    "read_file": read_file,
    "write_file": write_file,
    "search_code": search_code,
    "run_terminal": run_terminal,
    "apply_patch": apply_patch
}

memory = load_memory()

while True:

    task = input("task> ")

    steps = plan(task)

    for step in steps:

        tool = TOOLS[step["tool"]]
        result = tool(**step["args"])

        print(result)

    memory.append(task)
    save_memory(memory)

import os
import json

def build_index(root="src"):
    index = {}

    for path, _, files in os.walk(root):
        for f in files:
            if f.endswith(".py"):
                full = os.path.join(path, f)
                index[f] = full

    with open("index/symbol_index.json","w") as out:
        json.dump(index, out, indent=2)

    return index

7. Executor separato
agent/executor.py

from tools import *

TOOLS = {
 "read_file": read_file.run,
 "write_file": write_file.run,
 "search_code": search_code.run,
 "run_terminal": run_terminal.run,
 "apply_patch": apply_patch.run
}

def execute(step):

    tool = TOOLS.get(step["tool"])

    if not tool:
        return None

    args = step.get("args", {})

    return tool(**args)

4. Bridge principale
src/extension.ts

import * as vscode from "vscode"
import * as http from "http"

export function activate(context: vscode.ExtensionContext) {

    const server = http.createServer(async (req, res) => {

        if (req.method !== "POST") {
            res.end()
            return
        }

        let body = ""

        req.on("data", chunk => {
            body += chunk
        })

        req.on("end", async () => {

            const msg = JSON.parse(body)

            if (msg.action === "open_file") {

                const doc = await vscode.workspace.openTextDocument(msg.path)
                await vscode.window.showTextDocument(doc)

            }

            if (msg.action === "insert_text") {

                const editor = vscode.window.activeTextEditor

                if (editor) {
                    editor.edit(edit => {
                        edit.insert(
                            new vscode.Position(msg.line, msg.character),
                            msg.text
                        )
                    })
                }

            }

            if (msg.action === "replace_file") {

                const uri = vscode.Uri.file(msg.path)

                const edit = new vscode.WorkspaceEdit()

                const fullRange = new vscode.Range(
                    new vscode.Position(0,0),
                    new vscode.Position(100000,0)
                )

                edit.replace(uri, fullRange, msg.content)

                await vscode.workspace.applyEdit(edit)

            }

            if (msg.action === "run_terminal") {

                const terminal = vscode.window.createTerminal("OpenClaw")

                terminal.show()

                terminal.sendText(msg.command)

            }

            res.end("ok")

        })

    })

    server.listen(3210)

    console.log("OpenClaw bridge running on port 3210")
}

export function deactivate() {}

4. Costruire il grafo del codice
tools/build_graph.py

import os
import json
from tree_sitter import Parser

graph = {
    "functions": {},
    "classes": {},
    "calls": []
}

def index_file(path):

    with open(path) as f:
        code = f.read()

    # parsing semplificato
    lines = code.split("\n")

    for line in lines:

        if line.strip().startswith("def "):
            name = line.split("(")[0].replace("def","").strip()
            graph["functions"][name] = path

        if line.strip().startswith("class "):
            name = line.split(":")[0].replace("class","").strip()
            graph["classes"][name] = path

        if "(" in line and ")" in line:
            graph["calls"].append(line.strip())

def build(root="src"):

    for path,_,files in os.walk(root):
        for f in files:
            if f.endswith(".py"):
                index_file(os.path.join(path,f))

    with open("index/symbol_graph.json","w") as out:
        json.dump(graph,out,indent=2)

Perché Questo Approccio è Eccellente per i Tuoi Programmi

Il "ponte" basato su tool (CLI, file, server HTTP) è una forma di API (Application Programming Interface) per il tuo programma, pensata per essere usata da un agente AI. I vantaggi sono enormi:

Disaccoppiamento Totale (Decoupling): Il tuo programma e l'agente OpenClaw rimangono due entità separate.

Indipendenza tecnologica: Il tuo programma può essere scritto in Rust, Go, Java, C# o qualsiasi altro linguaggio. A OpenClaw non importa. L'agente interagisce solo con l'interfaccia (il tool), non con il codice interno del tuo programma.

Manutenzione semplificata: Puoi aggiornare e modificare il tuo programma senza dover toccare l'agente, purché l'interfaccia dei tool rimanga la stessa. E viceversa.

Interfaccia Controllata e Sicura: Sei tu a decidere quali azioni del tuo programma esporre all'agente. I tool diventano una "lista di permessi" (allowlist) di ciò che l'AI può fare. Se non crei un tool delete_database(), l'agente non potrà mai eseguire quell'azione.

Testabilità Semplificata: Puoi testare i tool e le interfacce del tuo programma in modo indipendente, senza nemmeno avviare l'agente AI. Questo rende il debugging mille volte più semplice.

Universalità: La comunicazione tramite riga di comando (CLI) o server HTTP locale è uno standard universale che funziona su qualsiasi sistema operativo e con qualsiasi stack tecnologico.

Questo è il metodo più stabile e disaccoppiato.

2. Approccio Server HTTP Locale (Come il bridge di VS Code)

Ideale per applicazioni che hanno uno stato interno complesso e che devono essere controllate in tempo reale.

Il tuo programma:

Integra un piccolo server web (con Flask/FastAPI in Python, Express in Node.js, etc.) che gira su una porta locale (es. localhost:12345).

Definisci degli endpoint per le azioni:

GET /status -> restituisce lo stato attuale.

POST /data -> aggiunge nuovi dati.

Il Tool per OpenClaw:

Usa una libreria come requests in Python per inviare richieste HTTP al tuo programma. È esattamente lo stesso pattern del bridge di VS Code.

3. Approccio Basato su File (Per Scambi di Dati Asincroni)

Utile se il tuo programma deve eseguire compiti lunghi e non vuoi che l'agente resti in attesa.

Funzionamento:

OpenClaw, tramite un tool, scrive un file di comando in una cartella specifica (es. project/input/command-1.json).

Il tuo programma "osserva" quella cartella. Quando trova un nuovo file, lo elabora.

Al termine, il tuo programma scrive un file di risultato in un'altra cartella (es. project/output/result-1.json).

Un altro tool di OpenClaw può controllare la cartella di output per recuperare i risultati.

Sceglieresti un Altro Approccio?

No, per il 99% dei casi, no.

L'alternativa principale sarebbe un'integrazione diretta tramite libreria, dove il tuo programma importa direttamente parti di OpenClaw (o viceversa). Questo è quasi sempre un errore perché crea un accoppiamento stretto (tight coupling):

Se aggiorni OpenClaw, potresti rompere il tuo programma.

Il tuo programma sarebbe vincolato allo stesso linguaggio di OpenClaw (Python).

La logica dell'agente e quella del tuo programma si mescolerebbero, rendendo tutto più difficile da gestire e testare.

In conclusione: l'architettura basata su "tool" che funge da "ponte" non è solo una soluzione per VS Code, ma è il paradigma di design corretto per costruire sistemi in cui un agente AI (come OpenClaw) interagisce in modo affidabile e controllato con altri software. È flessibile, robusto e scalabile.
